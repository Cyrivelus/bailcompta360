// Fonctions de gestion des emprunts (à implémenter dans fonctions/gestion_emprunts.php)

function ajouterEmprunt(
    $db,
    $banque,
    $dateSouscription,
    $agence,
    $devise,
    $numeroPret,
    $typePret,
    $client,
    $montantPret,
    $sourceFinancement,
    $tauxEffectifGlobal,
    $montantDiffere,
    $typePlan,
    $echeanceFinMois,
    $nombreEcheances,
    $gestionDiffere,
    $nombreJoursReels,
    $typeInteret,
    $dateMiseEnPlace,
    $datePremiereEcheance,
    $dateDerniereEcheance,
    $interetSPTaux,
    $interetSPMontant,
    $taxes,
    $periode,
    $terme,
    $perception,
    $dateDebutAmortissement,
    $dateFinAmortissement,
    $montantInitial,
    $duree,
    $typeAmortissement
) {
    try {
        $stmt = $db->prepare("INSERT INTO Emprunts_Bancaires (
            Banque, Date_Souscription, Agence, Devise, Numero_Pret, Type_Pret, Client, Montant_Pret,
            Source_Financement, Taux_Effectif_Global, Montant_Differe, Type_Plan, Echeance_Fin_Mois,
            Nombre_Echeances, Gestion_Differe, Nombre_Jours_Reels, Type_Interet, Date_Mise_En_Place,
            Date_Premiere_Echeance, Date_Derniere_Echeance, Interet_SP_Taux, Interet_SP_Montant,
            Taxes, Periode, Terme, Perception, Date_Debut_Amortissement, Date_Fin_Amortissement,
            Montant_Initial, Duree, Type_Amortissement
        ) VALUES (
            :banque, :date_souscription, :agence, :devise, :numero_pret, :type_pret, :client, :montant_pret,
            :source_financement, :taux_teg, :montant_differe, :type_plan, :echeance_fin_mois,
            :nombre_echeances, :gestion_differe, :nombre_jours_reels, :type_interet, :date_mise_en_place,
            :date_premiere_echeance, :date_derniere_echeance, :interet_sp_taux, :interet_sp_montant,
            :taxes, :periode, :terme, :perception, :date_debut_amortissement, :date_fin_amortissement,
            :montant_initial, :duree, :type_amortissement
        )");

        $stmt->bindParam(':banque', $banque);
        $stmt->bindParam(':date_souscription', $dateSouscription);
        $stmt->bindParam(':agence', $agence);
        $stmt->bindParam(':devise', $devise);
        $stmt->bindParam(':numero_pret', $numeroPret);
        $stmt->bindParam(':type_pret', $typePret);
        $stmt->bindParam(':client', $client);
        $stmt->bindParam(':montant_pret', $montantPret);
        $stmt->bindParam(':source_financement', $sourceFinancement);
        $stmt->bindParam(':taux_teg', $tauxEffectifGlobal);
        $stmt->bindParam(':montant_differe', $montantDiffere);
        $stmt->bindParam(':type_plan', $typePlan);
        $stmt->bindParam(':echeance_fin_mois', $echeanceFinMois, PDO::PARAM_BOOL);
        $stmt->bindParam(':nombre_echeances', $nombreEcheances, PDO::PARAM_INT);
        $stmt->bindParam(':gestion_differe', $gestionDiffere);
        $stmt->bindParam(':nombre_jours_reels', $nombreJoursReels, PDO::PARAM_INT);
        $stmt->bindParam(':type_interet', $typeInteret);
        $stmt->bindParam(':date_mise_en_place', $dateMiseEnPlace);
        $stmt->bindParam(':date_premiere_echeance', $datePremiereEcheance);
        $stmt->bindParam(':date_derniere_echeance', $dateDerniereEcheance);
        $stmt->bindParam(':interet_sp_taux', $interetSPTaux);
        $stmt->bindParam(':interet_sp_montant', $interetSPMontant);
        $stmt->bindParam(':taxes', $taxes);
        $stmt->bindParam(':periode', $periode);
        $stmt->bindParam(':terme', $terme);
        $stmt->bindParam(':perception', $perception);
        $stmt->bindParam(':date_debut_amortissement', $dateDebutAmortissement);
        $stmt->bindParam(':date_fin_amortissement', $dateFinAmortissement);
        $stmt->bindParam(':montant_initial', $montantInitial);
        $stmt->bindParam(':duree', $duree, PDO::PARAM_INT);
        $stmt->bindParam(':type_amortissement', $typeAmortissement);

        $stmt->execute();
        return $db->lastInsertId();
    } catch (PDOException $e) {
        error_log("Erreur lors de l'ajout de l'emprunt : " . $e->getMessage());return false;
    }
}
